*栈的链式的定义*

~~~cpp
typedef struct Stack Node{
    SElemType data;
    Struct StackNode *next;
}StackNode,*LinkStack;

LinkStack S;
~~~

*链栈的初始化*

~~~cpp
void InitStack(LinkStack S){
    if(S==NULL) return TRUE;
    else return FALSE;
}

~~~

*链栈的入栈*

~~~cpp
Status Push(LinkStack &S,SElemType e){
    p=new StackNode;//生成新节点p

    p->data=e;//将新节点数据域置尾e

    p->next=S;  //将新节点插入栈顶
  
    S=p; //修改栈顶指针

    return OK;

}
~~~
